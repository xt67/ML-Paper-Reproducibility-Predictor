3. Add new Test 9 for Full Pipeline:

```python
# Test 9: Full Pipeline Integration
print('9. Full Pipeline (classify + gaps + hints + explain)...')
try:
    from pathlib import Path
    
    model_path = Path('models/scibert_finetuned')
    if not model_path.exists():
        print('   ⚠ SKIPPED: No trained model (run training first)')
    else:
        from src.classifier import ReproducibilityClassifier
        from src.gap_detector import GapDetector
        from src.hint_generator import HintGenerator
        from src.explainer import SHAPExplainer
        
        # Sample methods text
        text = """
        We trained our model using the Adam optimizer with learning rate 0.001.
        Batch size was 32 and we trained for 100 epochs.
        Random seed was set to 42 for reproducibility.
        """
        
        # Classify
        clf = ReproducibilityClassifier(str(model_path))
        classification = clf.predict(text)
        
        # Detect gaps
        detector = GapDetector()
        gaps = detector.detect(text)
        summary = detector.summary(gaps)
        
        # Generate hints for first missing item (if any)
        missing = [g for g in gaps if g["status"] == "missing"]
        hint_gen = HintGenerator()
        if missing:
            hint = hint_gen.generate_hint(missing[0])
        else:
            hint = "No gaps found"
        
        # Explain
        explainer = SHAPExplainer(clf)
        explanation = explainer.explain(text)
        
        # Verify all components returned data
        assert 0 <= classification["score"] <= 1
        assert len(gaps) > 0
        assert isinstance(summary["coverage_score"], (int, float))
        assert len(explanation["sentences"]) > 0
        
        print(f'   ✓ Full pipeline works:')
        print(f'      Score: {classification["score"]:.3f}')
        print(f'      Gaps: {summary["missing"]} missing items')
        print(f'      Hint: "{hint[:40]}..."')
        print(f'      SHAP: {len(explanation["sentences"])} top sentences')
        passed += 1
except Exception as e:
    print(f'   ✗ FAILED: {e}')
    failed += 1
```
